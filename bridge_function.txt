// PHASE 1: BIO-DREAM GAUNTLET - Gray-Box Bridge
function createBridgeLevel() {
    clearLevel();
    
    const bridgeLength = 3500;
    const blockSpacing = 2;
    const blockSize = { width: 2, height: 1, depth: 2 };
    
    // Generate gray-box bridge blocks from X=0 to X=3500
    for (let x = 0; x < bridgeLength; x += blockSpacing) {
        const geometry = new THREE.BoxGeometry(blockSize.width, blockSize.height, blockSize.depth);
        const material = new THREE.MeshStandardMaterial({ 
            color: 0x808080, // Gray placeholder
            roughness: 0.7
        });
        const block = new THREE.Mesh(geometry, material);
        block.position.set(x, 0, 0);
        block.userData.isCollapsing = false;
        block.userData.collapseTimer = 0;
        block.userData.isFalling = false;
        scene.add(block);
        bridgeSegments.push(block);
    }
    
    // OBSTACLE 1: X=500 - Delete 3 blocks (3-unit gap for Base Jump - 10 ATP)
    const gap1Index = Math.floor(500 / blockSpacing);
    for (let i = 0; i < 3; i++) {
        const idx = gap1Index + i;
        if (bridgeSegments[idx]) {
            scene.remove(bridgeSegments[idx]);
            bridgeSegments[idx] = null;
        }
    }
    
    // OBSTACLE 2: X=1200 - Stack 5 wall blocks (Tier 1 Charge - 20 ATP)
    const wallX = 1200;
    for (let h = 1; h <= 5; h++) {
        const wallGeometry = new THREE.BoxGeometry(1, 5, 2);
        const wallMaterial = new THREE.MeshStandardMaterial({ 
            color: 0x666666,
            roughness: 0.8
        });
        const wallBlock = new THREE.Mesh(wallGeometry, wallMaterial);
        wallBlock.position.set(wallX, h * 2.5, 0);
        scene.add(wallBlock);
        bridgeSegments.push(wallBlock);
    }
    
    // OBSTACLE 3: X=2200 - Delete 8 blocks (massive gap for Jump + Dash - 25 ATP)
    const gap2Index = Math.floor(2200 / blockSpacing);
    for (let i = 0; i < 8; i++) {
        const idx = gap2Index + i;
        if (bridgeSegments[idx]) {
            scene.remove(bridgeSegments[idx]);
            bridgeSegments[idx] = null;
        }
    }
    
    // GOAL: X=3500 - Golden Cube (Placeholder for Rocket)
    const goalGeometry = new THREE.BoxGeometry(2, 2, 2);
    const goalMaterial = new THREE.MeshStandardMaterial({ 
        color: 0xFFD700, // Gold
        emissive: 0xFFAA00,
        emissiveIntensity: 0.5,
        metalness: 0.8
    });
    rocketModel = new THREE.Mesh(goalGeometry, goalMaterial);
    rocketModel.position.set(3500, 2, 0);
    scene.add(rocketModel);
    
    // Dark synaptic background
    scene.background = new THREE.Color(0x1a0a2e);
}
